-----------Creates necessarios para USLP07---------------

CREATE TABLE OperationScheduled (
    idOperation NUMBER PRIMARY KEY,
    operationName VARCHAR2(100),
    workstationId NUMBER,
    startTime NUMBER,
    endTime NUMBER,
    operationNumber NUMBER,
    articleId NUMBER
);


Criar a Tabela para os Tempos Médios de Produção:
CREATE TABLE production_times (
    articleId VARCHAR2(255) PRIMARY KEY,
    averageProductionTime NUMBER,
    averageWaitTime NUMBER
);
