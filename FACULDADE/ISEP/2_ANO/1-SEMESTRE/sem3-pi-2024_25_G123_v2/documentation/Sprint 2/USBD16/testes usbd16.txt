SET SERVEROUTPUT ON;

DECLARE
    v_success_message VARCHAR2(255);
BEGIN
    DBMS_OUTPUT.PUT_LINE('=== Test 1: Valid Insertion ===');
    v_success_message := RegisterProduct(
        p_product_name => 'Test Product 1',
        p_family_id => 1,
        p_product_description => 'Test product description'
    );
    DBMS_OUTPUT.PUT_LINE(v_success_message);

    DBMS_OUTPUT.PUT_LINE('=== Test 2: NULL Product Name ===');
    BEGIN
        v_success_message := RegisterProduct(
            p_product_name => NULL,
            p_family_id => 1,
            p_product_description => 'Test product description'
        );
        DBMS_OUTPUT.PUT_LINE(v_success_message);
    EXCEPTION
         WHEN OTHERS THEN
            DBMS_OUTPUT.PUT_LINE('Error: ' || SQLERRM);
        END;

    DBMS_OUTPUT.PUT_LINE('=== Test 3: Invalid Family ID ===');
    BEGIN
        v_success_message := RegisterProduct(
            p_product_name => 'Test Product 3',
            p_family_id => 9,
            p_product_description => 'Test product description'
        );
        DBMS_OUTPUT.PUT_LINE(v_success_message);
    EXCEPTION
        WHEN OTHERS THEN
            DBMS_OUTPUT.PUT_LINE('Error: ' || SQLERRM);
    END;
END;
/
